BDD employes

  • 1. Afficher le nom , le prénom et le service des employés qui ont une lettre e à la fin du prénom et travailent dans le service commercial ou comptabilité :
    SELECT * FROM employes WHERE prenom LIKE '%e' AND (service = "direction" OR service = "comptabilite");
  • 2. le nom , le prénom et la date d'embauche des employés qui ont été embaucher après 2007 et ont le sexe = 'f ' :
    SELECT nom, prenom, date_embauche FROM `employes` WHERE salaire > "2007-01-01" AND sex="f";
    ou
    SELECT nom, prenom, date_embauche FROM `employes` WHERE sex="f" AND dat_embauche BETWEEN '2022-01-01' AND CURDATE();
  • 3. Afficher les différents salaire touchées dans la base entreprise:

  • 4. Afficher les noms des employés qui ont un salaire entre 2300 et 4000 :
    SELECT nom FROM employes WHERE salaire BETWEEN 2300 AND 4000;
    ou
    SELECT nom FROM employes WHERE salaire >= 2300 AND salaire <= 4000;
  • 5. Afficher le nombre d'employés par salaire :
    ?????????????
    ou
    SELECT salaire, COUNT(*) AS nbr_employes FROM employes GROUPE BY salaire;
  • 6. Afficher la profession de l'employé 980 :
    SELECT service FROM `employes` WHERE id_employe = 980;
  • 7. Afficher la date d'embauche de Paul :
    SELECT date_embauche FROM `employes` WHERE prenom ="paul";
  • 8. Afficher le nombre de commerciaux :
    SELECT COUNT(*) AS nbrComerciaux FROM employes WHERE service= "commercial";
  • 9. Afficher le coût des commerciaux sur une année :
    SELECT SUM(salaire*12) FROM employes WHERE service = 'commercial';
  • 10. Afficher le salaire moyen par service :
    SELECT service, AVG(salaire) FROM employes GROUPE BY service;
  • 11. Afficher le nombre de recrutement sur l'année 2022 :
    SELECT COUNT(*) AS nbr_recutement FROM employes WHERE date_embauche BETWEEN '2022-01-01' AND '2022-12-31';
    ou
    SELECT COUNT(*) AS nbr_recutement FROM employes WHERE date_embauche >= '2022-01-01' AND dat_embauche <='2022-12-31';
    ou
    SELECT COUNT(*) AS nbr_recutement FROM employes WHERE date_embauche LIKE '%2022%'
  • 12. Augmenter le salaire pour chaque employé de 100€ :
    UPDATE employes SET salaire = salaire + 100;
  • 13. Afficher le nombre de service :
    SELECT COUNT(DISTINCT service) AS nbr_service FROM employes;
  • 14. Afficher 10 employés à partir de l'enregistrement 10 :
    SELECT * FROM `employes` LIMIT 10,10;
  • 15. Afficher les salaires qui sont touchées par plus qu'un employé :
    SELECT salaire FROM employes GROUPE BY salaire HAVING COUNT(*) > 1;

    -> CREATE TABLE stagiaires(
    id_stagiaire INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    prenom VARCHAR(100) NOT NULL,
    yeux VARCHAR(30) NOT NULL,
    genre enum('m', 'f') NOT NULL
    );
    -> create table bonbons(
    id_bonbon INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    nom VARCHAR(100) NOT NULL,
    saveur VARCHAR(100) NOT NULL
    );
    -> CREATE TABLE mange_bonbon(
    id_mange_bonbon INT(11) PRIMARY KEY AUTO_INCREMENT NOT NULL,
    id_bonbon INT(11) DEFAULT NULL,
    id_stagiaire INT(11) DEFAULT NULL,
    date_manger DATE NOT NULL,
    quantite INT(11) NOT NULL
    );
